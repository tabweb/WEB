
## 结构赋值
数组模型的结构
对象模型的结构

```js
// 数组模型结构赋值
{
    // 基本
    let [a, b, c] = [1, 2, 3];
    console.log(a, b, c);
}
{
    // 可嵌套
    let [a, [[b], c]] = [1, [[2], 3]];
    // a = 1
    // b = 2
    // c = 3
}
{
    // 可忽略
    let [a, , b] = [1, 2, 3];
    // a = 1
    // b = 3
}
{
    // 不完全解构
    let [a = 1, b] = []; // a = 1, b = undefined
}
{
    // 剩余运算符
    let [a, ...b] = [1, 2, 3, 4]
    //a = 1
    //b = [2, 3,4]
    console.log(a, b);
}
{
    // 字符串等
    // 在数组的解构中，解构的目标若为可遍历对象，皆可进行解构赋值。可遍历对象即实现 Iterator 接口的数据。
    let [a, b, c, d, e] = 'hello';
    // a = 'h'
    // b = 'e'
    // c = 'l'
    // d = 'l'
    // e = 'o'
    console.log(a, b, c, d, e)
}
{
    // 解构默认值
    let [a = 2] = [undefined]; // a = 2
    // 当解构模式有匹配结果，且匹配结果是 undefined 时，会触发默认值作为返回结果。
    let [a = 3, b = a] = [];     // a = 3, b = 3
    let [a = 3, b = a] = [1];    // a = 1, b = 1
    let [a = 3, b = a] = [1, 2]; // a = 1, b = 2
}
```
## 对象模型的结构
```js
{
	// 常用结构赋值
	let obj = {
		data:[1,2,3],
		pageNum:1,
	}
	let {data} = obj;
	console.log(data); // [1,2,3]
	
	//JavaScript引擎会将{a}理解成一个代码块
	//为了避免错误，不要讲{}写在行首
}
{
    // 基本
    let { foo, bar } = { foo: 'aaa', bar: 'bbb' };
    // foo = 'aaa'
    // bar = 'bbb'
    console.log(foo, bar)
    let { bar= foo } = { baz: 'sss' }
    // foo = 'ddd'
    console.log(baz)
}
{
    // 可嵌套可忽略
    let obj = { p: ['hello', { y: 'world' }] };
    let { p: [x, { y }] } = obj;
    // x = 'hello'
    // y = 'world'
    let obj = { p: ['hello', { y: 'world' }] };
    let { p: [x, { }] } = obj;
    // x = 'hello'
}
{
    // 不完全解构
    let obj = { p: [{ y: 'world' }] };
    let { p: [{ y }, x] } = obj;
    // x = undefined
    // y = 'world'
}
{
    // 剩余运算符
    let { a, b, ...rest } = { a: 10, b: 20, c: 30, d: 40 };
    // a = 10
    // b = 20
    // rest = {c: 30, d: 40}
}
{
    // 解构默认值
    let { a = 10, b = 5 } = { a: 3 };
    // a = 3; b = 5;
    let { a: aa = 10, b: bb = 5 } = { a: 3 };
    // aa = 3; bb = 5;
}

```
